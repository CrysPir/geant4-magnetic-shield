# Шаг 2   
  
Мы переходим ко второму шагу: добавление магнитного поля. В предыдущем файле DetectorConstruction была заглушка ConstructSDandField(). Теперь создадим отдельный класс MagneticField, который реализует формулу B = 2 - 1*r (Тесла, r в метрах), и подключим его к геометрии. После этого GEANT4 будет использовать это поле для расчёта траекторий заряженных частиц.  
  
Ниже — все необходимые файлы и изменения.  
  
---  
  
1. Создаём файл MagneticField.hh  
  
  
// MagneticField.hh  
#ifndef MagneticField_h  
#define MagneticField_h 1  
  
#include "G4MagneticField.hh"  
#include "G4SystemOfUnits.hh"  
  
class MagneticField : public G4MagneticField  
{  
public:  
    MagneticField();  
    virtual ~MagneticField();  
  
    // Основной метод, который GEANT4 вызывает для получения поля в точке  
    virtual void GetFieldValue(const double Point[4], double* Bfield) const;  
  
private:  
    // Параметры поля: B0 = 2 Тл на оси, градиент = 1 Тл/м  
    const double fB0 = 2.0 * tesla;  
    const double fGradient = 1.0 * tesla / m;  
};  
  
#endif  
  
  
---  
  
2. Создаём файл MagneticField.cc  
  
  
// MagneticField.cc  
#include "MagneticField.hh"  
#include <cmath>  
  
MagneticField::MagneticField()  
{}  
  
MagneticField::~MagneticField()  
{}  
  
void MagneticField::GetFieldValue(const double Point[4], double* Bfield) const  
{  
    // Point[0]=x, Point[1]=y, Point[2]=z (в метрах), Point[3]=t (время, не используется)  
    double x = Point[0];  
    double y = Point[1];  
    double r = std::sqrt(x*x + y*y);  // расстояние от оси Z  
  
    // Поле аксиально-симметричное: Bz = B0 - gradient * r  
    double Bz = fB0 - fGradient * r;  
    if (Bz < 0) Bz = 0;  // физически поле не становится отрицательным  
  
    // Компоненты поля: Bx и By = 0, Bz вычислено  
    Bfield[0] = 0.0;  
    Bfield[1] = 0.0;  
    Bfield[2] = Bz;  
}

void DetectorConstruction::ConstructSDandField()  
{  
    // 1. Создаём объект вашего поля  
    MagneticField* myField = new MagneticField();  
  
    // 2. Получаем глобальный FieldManager (управляет полем во всём мире)  
    G4FieldManager* fieldMgr = G4TransportationManager::GetTransportationManager()->GetFieldManager();  
  
    // 3. Устанавливаем поле  
    fieldMgr->SetDetectorField(myField);  
  
    // 4. Создаём уравнение движения (сила Лоренца)  
    G4Mag_UsualEqRhs* equation = new G4Mag_UsualEqRhs(myField);  
  
    // 5. Выбираем численный интегратор (RK4 — хорошая точность)  
    G4ClassicalRK4* stepper = new G4ClassicalRK4(equation);  
  
    // 6. Создаём ChordFinder для управления шагом интегрирования  
    //    Параметры: минимальный шаг (1e-3 мм), стэппер  
    G4ChordFinder* chordFinder = new G4ChordFinder(myField, 1.0e-3 * mm, stepper);  
    fieldMgr->SetChordFinder(chordFinder);  
  
    // 7. Дополнительные настройки точности (опционально)  
    fieldMgr->SetMinimumEpsilonStep(1.0e-5);  
    fieldMgr->SetDeltaIntersection(1.0e-4 * mm);  
}
